/**
 * main.c
 */

void main(void) { 
	WDTCTL = WDTPW | WDTHOLD;// Stop watchdog timer
 
	P1SEL =0; 
	P1DIR |=BIT0;         		//set Port 1.0 output 
	P5DIR &=~(BIT6);  		//set Port 5.6 input
	P5REN|=BIT6;			//enable pull-up resistor on 
	P1OUT|=BIT0;

	P1IE |=BIT0;//enable the interrupt on Port 1.1 
	P1IES |=BIT0;//set as falling edge 
	P1IFG &=~(BIT1);//clear interrupt flag
 
	//enter LPM4 mode and enable interrupt 
	_BIS_SR(LPM4_bits + GIE); 
	} 
	
	//Port 1 ISR 
#pragma vector=PORT1_VECTOR 
	_interrupt void PORT_1(void) 
	{ 
	P1OUT ^=0x01; 
	P1IFG &=~(BIT1);
	}
